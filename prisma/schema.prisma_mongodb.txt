// Prisma schema for MongoDB

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "mongodb"
  url      = env("DATABASE_URL")
}

// Enums
enum AppRole {
  admin
  researcher
  moderator
  viewer
}

enum VerdictStatus {
  true
  false
  pending
  inconclusive
}

enum ClaimCategory {
  Health
  Politics
  Economy
  Security
  Others
}

enum EvidenceType {
  image
  video
  link
  document
}

enum NotificationType {
  comment
  like
  verdict
  appeal
  mention
}

enum ReportReason {
  spam
  misleading
  inappropriate
  other
}

// Models
model State {
  id        String   @id @default(auto()) @map("_id") @db.ObjectId
  name      String   @unique
  createdAt DateTime @default(now()) @map("created_at")

  lgas      LGA[]

  @@map("states")
}

model LGA {
  id         String   @id @default(auto()) @map("_id") @db.ObjectId
  name       String
  stateId    String   @map("state_id") @db.ObjectId
  createdAt  DateTime @default(now()) @map("created_at")

  state      State    @relation(fields: [stateId], references: [id], onDelete: Cascade)
  claims     Claim[]

  @@unique([name, stateId])
  @@map("lgas")
}

model Profile {
  id             String   @id @default(auto()) @map("_id") @db.ObjectId
  fullName       String?
  avatarUrl      String?
  bio            String?
  location       String?
  verifiedClaims Int      @default(0)
  createdAt      DateTime @default(now())
  updatedAt      DateTime @default(now()) @updatedAt

  @@map("profiles")
}

model UserRole {
  id        String   @id @default(auto()) @map("_id") @db.ObjectId
  userId    String
  role      AppRole  @default(viewer)
  createdAt DateTime @default(now())

  @@unique([userId, role])
  @@map("user_roles")
}

model Claim {
  id                String         @id @default(auto()) @map("_id") @db.ObjectId
  userId            String         @map("user_id") @db.ObjectId
  title             String
  description       String
  lgaId             String         @map("lga_id") @db.ObjectId
  category          ClaimCategory
  verdict           VerdictStatus  @default(pending)
  verdictSummary    String?        @map("verdict_summary")
  submittedDate     DateTime       @default(now()) @map("submitted_date")
  verdictDate       DateTime?      @map("verdict_date")
  investigatorNotes String?        @map("investigator_notes")
  investigatorId    String?        @map("investigator_id") @db.ObjectId
  views             Int            @default(0)
  isAnonymous       Boolean        @default(false) @map("is_anonymous")
  createdAt         DateTime       @default(now()) @map("created_at")
  updatedAt         DateTime       @default(now()) @updatedAt @map("updated_at")
  
  lga               LGA            @relation(fields: [lgaId], references: [id])
  evidence          Evidence[]
  citations         Citation[]
  comments          Comment[]
  likes             Like[]
  shares            Share[]
  appeals           Appeal[]
  reports           Report[]
  relatedFrom       RelatedClaim[] @relation("ClaimToRelated")
  relatedTo         RelatedClaim[] @relation("RelatedToClaim")

  @@map("claims")
}

model Evidence {
  id        String   @id @default(auto()) @map("_id") @db.ObjectId
  claimId   String   @map("claim_id") @db.ObjectId
  type      EvidenceType
  url       String
  caption   String?
  createdAt DateTime @default(now())

  claim     Claim?   @relation(fields: [claimId], references: [id])

  @@map("evidence")
}

model Citation {
  id        String   @id @default(auto()) @map("_id") @db.ObjectId
  claimId   String   @map("claim_id") @db.ObjectId
  title     String
  url       String
  createdAt DateTime @default(now())

  claim     Claim?   @relation(fields: [claimId], references: [id])

  @@map("citations")
}

model Comment {
  id        String    @id @default(auto()) @map("_id") @db.ObjectId
  claimId   String    @map("claim_id") @db.ObjectId
  userId    String    @map("user_id") @db.ObjectId
  content   String
  parentId  String?   @map("parent_id") @db.ObjectId
  createdAt DateTime  @default(now()) @map("created_at")
  updatedAt DateTime  @default(now()) @updatedAt @map("updated_at")

  claim     Claim     @relation(fields: [claimId], references: [id], onDelete: Cascade)
  parent    Comment?  @relation("CommentReplies", fields: [parentId], references: [id], onDelete: NoAction, onUpdate: NoAction)
  replies   Comment[] @relation("CommentReplies")

  @@map("comments")
}

model Like {
  id        String   @id @default(auto()) @map("_id") @db.ObjectId
  claimId   String   @map("claim_id") @db.ObjectId
  userId    String
  createdAt DateTime @default(now())

  claim     Claim?   @relation(fields: [claimId], references: [id])

  @@unique([claimId, userId])
  @@map("likes")
}

model Share {
  id        String   @id @default(auto()) @map("_id") @db.ObjectId
  claimId   String   @map("claim_id") @db.ObjectId
  userId    String
  platform  String?
  createdAt DateTime @default(now())

  claim     Claim?   @relation(fields: [claimId], references: [id])

  @@map("shares")
}

model Notification {
  id        String           @id @default(auto()) @map("_id") @db.ObjectId
  userId    String
  type      NotificationType
  title     String
  message   String
  link      String?
  read      Boolean          @default(false)
  createdAt DateTime         @default(now())

  @@map("notifications")
}

model Appeal {
  id            String   @id @default(auto()) @map("_id") @db.ObjectId
  claimId       String   @map("claim_id") @db.ObjectId
  userId        String
  reason        String
  status        String   @default("pending")
  adminResponse String?
  createdAt     DateTime @default(now())
  updatedAt     DateTime @default(now()) @updatedAt

  claim         Claim?   @relation(fields: [claimId], references: [id])

  @@map("appeals")
}

model Report {
  id            String       @id @default(auto()) @map("_id") @db.ObjectId
  claimId       String       @map("claim_id") @db.ObjectId
  userId        String
  reason        ReportReason
  details       String?
  status        String       @default("pending")
  adminResponse String?
  createdAt     DateTime     @default(now())
  updatedAt     DateTime     @default(now()) @updatedAt

  claim         Claim?       @relation(fields: [claimId], references: [id])

  @@map("reports")
}

model RelatedClaim {
  id             String   @id @default(auto()) @map("_id") @db.ObjectId
  claimId        String   @map("claim_id") @db.ObjectId
  relatedClaimId String   @map("related_claim_id") @db.ObjectId
  createdAt      DateTime @default(now())

  claim          Claim?   @relation("ClaimToRelated", fields: [claimId], references: [id])
  relatedClaim   Claim?   @relation("RelatedToClaim", fields: [relatedClaimId], references: [id])

  @@unique([claimId, relatedClaimId])
  @@map("related_claims")
}
